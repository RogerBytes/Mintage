En dépendances
Installer rust
https://www.rust-lang.org/tools/install
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh
et sqlite3, openssl
sudo nala install -y sqlite3 libsqlite3-dev openssl libssl-dev libgdk-pixbuf2.0-dev libgtk-4-dev libxml2-dev libgmock-dev googletest build-essential dpkg-dev devscripts

mettre à jour meson
wget https://github.com/mesonbuild/meson/releases/download/1.3.1/meson-1.3.1.tar.gz
tar -xzf meson*.tar.gz
cd meson-*/
sudo pip3 install .
cd
supprimer la merde du home manuellement le tar gz et le dossier meson je veux dire

Ensuite compiler
REPO_URL=https://gitlab.gnome.org/World/Shortwave
REPO_NAME=$(basename $REPO_URL .git)
SOURCE_DIR=$(pwd)/$REPO_NAME
git clone $REPO_URL $SOURCE_DIR
cd $SOURCE_DIR
# Les commandes ici
meson --prefix=/usr build
ninja -C build
sudo ninja -C build install
cd
sudo rm -r $SOURCE_DIR
